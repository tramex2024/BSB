const mongoose = require('mongoose');

const AIBotOrderSchema = new mongoose.Schema({
    symbol: { type: String, default: 'BTC_USDT' },
    side: { type: String, enum: ['BUY', 'SELL'] },
    price: Number,
    amount: Number,
    isVirtual: { type: Boolean, default: true }, // Paso 5 de tu plan
    confidenceScore: Number, // Qu√© tan segura estaba la IA (0-100)
    status: { type: String, default: 'FILLED' },
    timestamp: { type: Date, default: Date.now }
});

module.exports = mongoose.model('AIBotOrder', AIBotOrderSchema);
