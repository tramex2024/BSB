// server/routes/ordersRoutes.js

const express = require('express');
const router = express.Router();
const orderController = require('../controllers/orderController'); 
const authMiddleware = require('../middleware/authMiddleware'); 

// Esta ruta acepta cualquier "status" como parámetro dinámico (opened, filled, cancelled, all)
// La ruta completa es: GET /api/orders/:status
router.get('/:status', authMiddleware, orderController.getOrders);

module.exports = router;
